1#include "ImGui/imgui.h"
2// Mini memory editor for Dear ImGui (to embed in your game/tools)
3// Get latest version at http://www.github.com/ocornut/imgui_club
4//
5// Right-click anywhere to access the Options menu!
6// You can adjust the keyboard repeat delay/rate in ImGuiIO.
7// The code assume a mono-space font for simplicity!
8// If you don't use the default font, use ImGui::PushFont()/PopFont() to switch to a mono-space font before calling this.
9//
10// Usage:
11// // Create a window and draw memory editor inside it:
12// static MemoryEditor mem_edit_1;
13// static char data[0x10000];
14// size_t data_size = 0x10000;
15// mem_edit_1.DrawWindow("Memory Editor", data, data_size);
16//
17// Usage:
18// // If you already have a window, use DrawContents() instead:
19// static MemoryEditor mem_edit_2;
20// ImGui::Begin("MyWindow")
21// mem_edit_2.DrawContents(this, sizeof(*this), (size_t)this);
22// ImGui::End();
23//

163 void CalcSizes(Sizes& s, size_t mem_size, size_t base_display_addr)
164 {
165 ImGuiStyle& style = ImGui::GetStyle();
166 s.AddrDigitsCount = OptAddrDigitsCount;
167 if (s.AddrDigitsCount == 0)
168 for (size_t n = base_display_addr + mem_size - 1; n > 0; n >>= 4)
169 s.AddrDigitsCount++;
170 s.LineHeight = ImGui::GetTextLineHeight();
171 s.GlyphWidth = ImGui::CalcTextSize("F").x + 1; // We assume the font is mono-space
172 s.HexCellWidth = (float)(int)(s.GlyphWidth * 2.5f); // "FF " we include trailing space in the width to easily catch clicks everywhere
173 s.SpacingBetweenMidCols = (float)(int)(s.HexCellWidth * 0.25f); // Every OptMidColsCount columns we add a bit of extra spacing
174 s.PosHexStart = (s.AddrDigitsCount + 2) * s.GlyphWidth;
175 s.PosHexEnd = s.PosHexStart + (s.HexCellWidth * Cols);
176 s.PosAsciiStart = s.PosAsciiEnd = s.PosHexEnd;
177 if (OptShowAscii)
178 {
179 s.PosAsciiStart = s.PosHexEnd + s.GlyphWidth * 1;
180 if (OptMidColsCount > 0)
181 s.PosAsciiStart += (float)((Cols + OptMidColsCount - 1) / OptMidColsCount) * s.SpacingBetweenMidCols;
182 s.PosAsciiEnd = s.PosAsciiStart + Cols * s.GlyphWidth;
183 }
184 s.WindowWidth = s.PosAsciiEnd + style.ScrollbarSize + style.WindowPadding.x * 2 + s.GlyphWidth;
185 }

256 else if (ImGui::IsKeyPressed(ImGui::GetKeyIndex(ImGuiKey_RightArrow)) && (ptrdiff_t)DataEditingAddr < (ptrdiff_t)mem_size - 1) { data_editing_addr_next = DataEditingAddr + 1; }

262 draw_list->AddLine(ImVec2(window_pos.x + s.PosAsciiStart - s.GlyphWidth, window_pos.y), ImVec2(window_pos.x + s.PosAsciiStart - s.GlyphWidth, window_pos.y + 9999), ImGui::GetColorU32(ImGuiCol_Border));
263

267 const char* format_address = OptUpperCaseHex ? "%0*" _PRISizeT "X: " : "%0*" _PRISizeT "x: ";

281 float byte_pos_x = s.PosHexStart + s.HexCellWidth * n;
282 if (OptMidColsCount > 0)
283 byte_pos_x += (float)(n / OptMidColsCount) * s.SpacingBetweenMidCols;
284 ImGui::SameLine(byte_pos_x);

286 // Draw highlight
287 bool is_highlight_from_user_range = (addr >= HighlightMin && addr < HighlightMax);
288 bool is_highlight_from_user_func = (HighlightFn && HighlightFn(mem_data, addr));
289 bool is_highlight_from_preview = (addr >= DataPreviewAddr && addr < DataPreviewAddr + preview_data_type_size);
290 if (is_highlight_from_user_range || is_highlight_from_user_func || is_highlight_from_preview)
291 {
292 ImVec2 pos = ImGui::GetCursorScreenPos();
293 float highlight_width = s.GlyphWidth * 2;
294 bool is_next_byte_highlighted = (addr + 1 < mem_size) && ((HighlightMax != (size_t)-1 && addr + 1 < HighlightMax) || (HighlightFn && HighlightFn(mem_data, addr + 1)));
295 if (is_next_byte_highlighted || (n + 1 == Cols))
296 {
297 highlight_width = s.HexCellWidth;
298 if (OptMidColsCount > 0 && n > 0 && (n + 1) < Cols && ((n + 1) % OptMidColsCount) == 0)
299 highlight_width += s.SpacingBetweenMidCols;

302 }

305 {
306 // Display text input on current byte
307 bool data_write = false;
308 ImGui::PushID((void*)addr);

311 ImGui::SetKeyboardFocusHere(0);
312 sprintf(AddrInputBuf, format_data, s.AddrDigitsCount, base_display_addr + addr);
313 sprintf(DataInputBuf, format_byte, ReadFn ? ReadFn(mem_data, addr) : mem_data[addr]);

315 struct UserData
316 {
317 // FIXME: We should have a way to retrieve the text edit cursor position more easily in the API, this is rather tedious. This is such a ugly mess we may be better off not using InputText() at all here.
318 static int Callback(ImGuiInputTextCallbackData* data)
319 {
320 UserData* user_data = (UserData*)data->UserData;
321 if (!data->HasSelection())
322 user_data->CursorPos = data->CursorPos;
323 if (data->SelectionStart == 0 && data->SelectionEnd == data->BufTextLen)
324 {
325 // When not editing a byte, always refresh its InputText content pulled from underlying memory data
326 // (this is a bit tricky, since InputText technically "owns" the master copy of the buffer we edit it in there)
327 data->DeleteChars(0, data->BufTextLen);
328 data->InsertChars(0, user_data->CurrentBufOverwrite);
329 data->SelectionStart = 0;
330 data->SelectionEnd = 2;
331 data->CursorPos = 0;
332 }
333 return 0;
334 }
335 char CurrentBufOverwrite[3]; // Input
336 int CursorPos; // Output
337 };
338 UserData user_data;
339 user_data.CursorPos = -1;
340 sprintf(user_data.CurrentBufOverwrite, format_byte, ReadFn ? ReadFn(mem_data, addr) : mem_data[addr]);
341 ImGuiInputTextFlags flags = ImGuiInputTextFlags_CharsHexadecimal | ImGuiInputTextFlags_EnterReturnsTrue | ImGuiInputTextFlags_AutoSelectAll | ImGuiInputTextFlags_NoHorizontalScroll | ImGuiInputTextFlags_CallbackAlways;
342#if IMGUI_VERSION_NUM >= 18104
343 flags |= ImGuiInputTextFlags_AlwaysOverwrite;
344#else
345 flags |= ImGuiInputTextFlags_AlwaysInsertMode;
346#endif
347 ImGui::SetNextItemWidth(s.GlyphWidth * 2);
348 if (ImGui::InputText("##data", DataInputBuf, IM_ARRAYSIZE(DataInputBuf), flags, UserData::Callback, &user_data))
349 data_write = data_next = true;
350 else if (!DataEditingTakeFocus && !ImGui::IsItemActive())
351 DataEditingAddr = data_editing_addr_next = (size_t)-1;
352 DataEditingTakeFocus = false;
353 if (user_data.CursorPos >= 2)
354 data_write = data_next = true;
355 if (data_editing_addr_next != (size_t)-1)
356 data_write = data_next = false;
357 unsigned int data_input_value = 0;
358 if (data_write && sscanf(DataInputBuf, "%X", &data_input_value) == 1)
359 {
360 if (WriteFn)
361 WriteFn(mem_data, addr, (ImU8)data_input_value);
362 else
363 mem_data[addr] = (ImU8)data_input_value;
364 }
365 ImGui::PopID();
366 }
367 else
368 {
369 // NB: The trailing space is not visible but ensure there's no gap that the mouse cannot click on.
370 ImU8 b = ReadFn ? ReadFn(mem_data, addr) : mem_data[addr];
371
372 if (OptShowHexII)
373 {
374 if ((b >= 32 && b < 128))
375 ImGui::Text(".%c ", b);
376 else if (b == 0xFF && OptGreyOutZeroes)
377 ImGui::TextDisabled("## ");
378 else if (b == 0x00)
379 ImGui::Text(" ");
380 else
381 ImGui::Text(format_byte_space, b);
382 }
383 else
384 {
385 if (b == 0 && OptGreyOutZeroes)
386 ImGui::TextDisabled("00 ");
387 else
388 ImGui::Text(format_byte_space, b);
389 }
390 if (!ReadOnly && ImGui::IsItemHovered() && ImGui::IsMouseClicked(0))
391 {
392 DataEditingTakeFocus = true;
393 data_editing_addr_next = addr;
394 }
395 }
396 }
397
398 if (OptShowAscii)
399 {
400 // Draw ASCII values
401 ImGui::SameLine(s.PosAsciiStart);
402 ImVec2 pos = ImGui::GetCursorScreenPos();
403 addr = line_i * Cols;
404 ImGui::PushID(line_i);
405 if (ImGui::InvisibleButton("ascii", ImVec2(s.PosAsciiEnd - s.PosAsciiStart, s.LineHeight)))
406 {
407 DataEditingAddr = DataPreviewAddr = addr + (size_t)((ImGui::GetIO().MousePos.x - pos.x) / s.GlyphWidth);
408 DataEditingTakeFocus = true;
409 }
410 ImGui::PopID();
411 for (int n = 0; n < Cols && addr < mem_size; n++, addr++)
412 {
413 if (addr == DataEditingAddr)
414 {
415 draw_list->AddRectFilled(pos, ImVec2(pos.x + s.GlyphWidth, pos.y + s.LineHeight), ImGui::GetColorU32(ImGuiCol_FrameBg));
416 draw_list->AddRectFilled(pos, ImVec2(pos.x + s.GlyphWidth, pos.y + s.LineHeight), ImGui::GetColorU32(ImGuiCol_TextSelectedBg));
417 }
418 unsigned char c = ReadFn ? ReadFn(mem_data, addr) : mem_data[addr];
419 char display_c = (c < 32 || c >= 128) ? '.' : c;
420 draw_list->AddText(pos, (display_c == c) ? color_text : color_disabled, &display_c, &display_c + 1);
421 pos.x += s.GlyphWidth;
422 }
423 }
424 }
425 ImGui::PopStyleVar(2);
426 ImGui::EndChild();
427
428 // Notify the main window of our ideal child content size (FIXME: we are missing an API to get the contents size from the child)
429 ImGui::SetCursorPosX(s.WindowWidth);
430
431 if (data_next && DataEditingAddr + 1 < mem_size)
432 {
433 DataEditingAddr = DataPreviewAddr = DataEditingAddr + 1;
434 DataEditingTakeFocus = true;
435 }
436 else if (data_editing_addr_next != (size_t)-1)
437 {
438 DataEditingAddr = DataPreviewAddr = data_editing_addr_next;
439 DataEditingTakeFocus = true;
440 }
441
442 const bool lock_show_data_preview = OptShowDataPreview;
443 if (OptShowOptions)
444 {
445 ImGui::Separator();

447 }

571 static void* EndianessCopyBigEndian(void* _dst, void* _src, size_t s, int is_little_endian)
572 {
573 if (is_little_endian)
574 {
575 uint8_t* dst = (uint8_t*)_dst;
576 uint8_t* src = (uint8_t*)_src + s - 1;
577 for (int i = 0, n = (int)s; i < n; ++i)
578 memcpy(dst++, src--, 1);
579 return _dst;
580 }
581 else
582 {
583 return memcpy(_dst, _src, s);
584 }
585 }
586
587 static void* EndianessCopyLittleEndian(void* _dst, void* _src, size_t s, int is_little_endian)
588 {
589 if (is_little_endian)
590 {
591 return memcpy(_dst, _src, s);
592 }
593 else
594 {
595 uint8_t* dst = (uint8_t*)_dst;
596 uint8_t* src = (uint8_t*)_src + s - 1;
597 for (int i = 0, n = (int)s; i < n; ++i)
598 memcpy(dst++, src--, 1);
599 return _dst;
600 }
601 }

604 {
605 static void* (*fp)(void*, void*, size_t, int) = NULL;
606 if (fp == NULL)
607 fp = IsBigEndian() ? EndianessCopyBigEndian : EndianessCopyLittleEndian;
608 return fp(dst, src, size, PreviewEndianess);

611 const char* FormatBinary(const uint8_t* buf, int width) const
612 {
613 IM_ASSERT(width <= 64);
614 size_t out_n = 0;
615 static char out_buf[64 + 8 + 1];
616 int n = width / 8;
617 for (int j = n - 1; j >= 0; --j)
618 {
619 for (int i = 0; i < 8; ++i)
620 out_buf[out_n++] = (buf[j] & (1 << (7 - i))) ? '1' : '0';
621 out_buf[out_n++] = ' ';
622 }
623 IM_ASSERT(out_n < IM_ARRAYSIZE(out_buf));
624 out_buf[out_n] = 0;
625 return out_buf;
626 }

630 {

634 if (ReadFn)

640 if (data_format == DataFormat_Bin)

642 uint8_t binbuf[8];
643 EndianessCopy(binbuf, buf, size);
644 ImSnprintf(out_buf, out_buf_size, "%s", FormatBinary(binbuf, (int)size * 8));
645 return;
646 }
647
648 out_buf[0] = 0;
649 switch (data_type)
650 {
651 case ImGuiDataType_S8:
652 {
653 int8_t int8 = 0;
654 EndianessCopy(&int8, buf, size);

656 if (data_format == DataFormat_Hex) { ImSnprintf(out_buf, out_buf_size, "0x%02x", int8 & 0xFF); return; }

668 {

672 if (data_format == DataFormat_Hex) { ImSnprintf(out_buf, out_buf_size, "0x%04x", int16 & 0xFFFF); return; }

675 case ImGuiDataType_U16:
676 {
677 uint16_t uint16 = 0;

679 if (data_format == DataFormat_Dec) { ImSnprintf(out_buf, out_buf_size, "%hu", uint16); return; }
680 if (data_format == DataFormat_Hex) { ImSnprintf(out_buf, out_buf_size, "0x%04x", uint16 & 0xFFFF); return; }
